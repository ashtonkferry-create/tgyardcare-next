Build TG-102 as the weekly aggregation workflow that compiles all SEO data into one email. Runs Monday 8 AM CT (after TG-97/98/99 have completed their Monday runs).

Workflow structure:
```
Schedule Trigger (cron: 0 13 * * 1 = Monday 8 AM CT)

  -> Code Node "Calculate Date Ranges"
     const now = new Date();
     const thisWeekEnd = new Date(now.getTime() - 2*24*60*60*1000).toISOString().split('T')[0];
     const thisWeekStart = new Date(now.getTime() - 9*24*60*60*1000).toISOString().split('T')[0];
     const lastWeekEnd = new Date(now.getTime() - 9*24*60*60*1000).toISOString().split('T')[0];
     const lastWeekStart = new Date(now.getTime() - 16*24*60*60*1000).toISOString().split('T')[0];
     // Monday of this week for seo_weekly_reports key
     const monday = new Date(now);
     monday.setDate(monday.getDate() - monday.getDay() + 1);
     const weekStart = monday.toISOString().split('T')[0];
     return [{ json: { thisWeekStart, thisWeekEnd, lastWeekStart, lastWeekEnd, weekStart } }];

  -> HTTP Request "This Week Queries" (parallel)
     POST GSC searchAnalytics/query
     Body: { startDate, endDate: thisWeekEnd, dimensions: ["query"], rowLimit: 25000 }

  -> HTTP Request "Last Week Queries" (parallel)
     POST GSC searchAnalytics/query
     Body: { startDate: lastWeekStart, endDate: lastWeekEnd, dimensions: ["query"], rowLimit: 25000 }

  -> HTTP Request "This Week Pages" (parallel)
     POST GSC searchAnalytics/query
     Body: { startDate: thisWeekStart, endDate: thisWeekEnd, dimensions: ["page"], rowLimit: 25000 }

  -> HTTP Request "Get Content Gaps Count"
     GET .../seo_content_gaps?status=eq.new&select=id
     Headers: Prefer: count=exact
     fullResponse: true

  -> HTTP Request "Get Recent Rank Drop Alerts"
     // Get TG-97 alert data stored this week
     GET .../seo_weekly_reports?week_start=eq.{{ weekStart }}&select=report_data
     fullResponse: true

  -> Code Node "Compile Summary"
     const thisWeek = ($('This Week Queries').first().json.rows || []);
     const lastWeek = ($('Last Week Queries').first().json.rows || []);
     const thisWeekPages = ($('This Week Pages').first().json.rows || []);

     // Parse content gaps count from content-range header
     const gapCountHeader = $('Get Content Gaps Count').first().json.headers?.['content-range'] || '*/0';
     const gapCount = parseInt(gapCountHeader.split('/')[1] || '0');

     // Parse rank drop data from seo_weekly_reports
     const reportData = $('Get Recent Rank Drop Alerts').first().json.body?.[0]?.report_data || {};

     // Build last week query map
     const lastMap = new Map(lastWeek.map(q => [q.keys[0], q]));

     // Top gainers (position improved most)
     const gainers = [];
     const losers = [];
     const newKeywords = [];
     const lostKeywords = [];

     for (const cur of thisWeek) {
       const query = cur.keys[0];
       const prev = lastMap.get(query);
       if (!prev) {
         if (cur.impressions >= 5) newKeywords.push({ query, impressions: cur.impressions, position: Math.round(cur.position) });
         continue;
       }
       const change = prev.position - cur.position; // positive = improved
       if (change >= 3) gainers.push({ query, change: Math.round(change), position: Math.round(cur.position), impressions: cur.impressions });
       if (change <= -3) losers.push({ query, change: Math.round(change), position: Math.round(cur.position), impressions: cur.impressions });
       lastMap.delete(query);
     }

     // Remaining in lastMap = lost keywords
     for (const [query, data] of lastMap) {
       if (data.impressions >= 5) lostKeywords.push({ query, lastImpressions: data.impressions, lastPosition: Math.round(data.position) });
     }

     // Sort by impact
     gainers.sort((a, b) => b.impressions * b.change - a.impressions * a.change);
     losers.sort((a, b) => a.change * a.impressions - b.change * b.impressions);

     // Overall stats
     const totalClicks = thisWeek.reduce((s, q) => s + q.clicks, 0);
     const totalImpressions = thisWeek.reduce((s, q) => s + q.impressions, 0);
     const prevTotalClicks = lastWeek.reduce((s, q) => s + q.clicks, 0);
     const prevTotalImpressions = lastWeek.reduce((s, q) => s + q.impressions, 0);

     return [{ json: {
       weekStart: $('Calculate Date Ranges').first().json.weekStart,
       overview: {
         totalClicks, prevTotalClicks,
         clickChange: totalClicks - prevTotalClicks,
         totalImpressions, prevTotalImpressions,
         impressionChange: totalImpressions - prevTotalImpressions,
         totalPages: thisWeekPages.length,
         totalQueries: thisWeek.length
       },
       topGainers: gainers.slice(0, 10),
       topLosers: losers.slice(0, 10),
       newKeywords: newKeywords.sort((a, b) => b.impressions - a.impressions).slice(0, 10),
       lostKeywords: lostKeywords.sort((a, b) => b.lastImpressions - a.lastImpressions).slice(0, 10),
       rankDropAlerts: reportData.rank_drops || [],
       ctrAnomalies: reportData.ctr_anomalies || [],
       contentGapCount: gapCount
     }}];

  -> HTTP Request "Store Weekly Report"
     POST https://lwtmvzhwekgdxkaisfra.supabase.co/rest/v1/seo_weekly_reports
     Headers: Prefer: resolution=merge-duplicates,return=minimal
     Body: {
       "week_start": "{{ weekStart }}",
       "report_data": {{ full summary JSON }},
       "top_gaining_keywords": {{ topGainers.map(g => g.query) }},
       "top_losing_keywords": {{ topLosers.map(l => l.query) }},
       "new_keywords": {{ newKeywords.map(n => n.query) }}
     }

  -> Code Node "Build Summary Email"
     const d = $json;
     const o = d.overview;

     // Helper for change indicators
     const arrow = (val) => val > 0 ? '&#9650;' : val < 0 ? '&#9660;' : '&#8212;';
     const color = (val) => val > 0 ? 'green' : val < 0 ? 'red' : 'gray';

     let html = `
     <h1>Weekly SEO Summary</h1>
     <p>Week of ${d.weekStart}</p>

     <h2>Overview</h2>
     <table border="1" cellpadding="8">
       <tr><th>Metric</th><th>This Week</th><th>Last Week</th><th>Change</th></tr>
       <tr><td>Total Clicks</td><td>${o.totalClicks}</td><td>${o.prevTotalClicks}</td>
           <td style="color:${color(o.clickChange)}">${arrow(o.clickChange)} ${o.clickChange}</td></tr>
       <tr><td>Total Impressions</td><td>${o.totalImpressions}</td><td>${o.prevTotalImpressions}</td>
           <td style="color:${color(o.impressionChange)}">${arrow(o.impressionChange)} ${o.impressionChange}</td></tr>
       <tr><td>Pages Tracked</td><td colspan="3">${o.totalPages}</td></tr>
       <tr><td>Queries Tracked</td><td colspan="3">${o.totalQueries}</td></tr>
     </table>`;

     // Rank drop alerts (if any from TG-97)
     if (d.rankDropAlerts.length > 0) {
       html += '<h2 style="color:red">Rank Drop Alerts</h2><ul>';
       d.rankDropAlerts.forEach(a => { html += `<li>"${a.query}" dropped ${a.drop} positions (${a.oldPosition} -> ${a.newPosition})</li>`; });
       html += '</ul>';
     }

     // Top gainers
     if (d.topGainers.length > 0) {
       html += '<h2 style="color:green">Top Gainers</h2><table border="1" cellpadding="6"><tr><th>Query</th><th>Position Change</th><th>Current Position</th></tr>';
       d.topGainers.forEach(g => { html += `<tr><td>${g.query}</td><td style="color:green">+${g.change}</td><td>${g.position}</td></tr>`; });
       html += '</table>';
     }

     // Top losers
     if (d.topLosers.length > 0) {
       html += '<h2 style="color:red">Top Losers</h2><table border="1" cellpadding="6"><tr><th>Query</th><th>Position Change</th><th>Current Position</th></tr>';
       d.topLosers.forEach(l => { html += `<tr><td>${l.query}</td><td style="color:red">${l.change}</td><td>${l.position}</td></tr>`; });
       html += '</table>';
     }

     // New and lost keywords
     if (d.newKeywords.length > 0) {
       html += '<h2>New Keywords</h2><ul>';
       d.newKeywords.forEach(k => { html += `<li>"${k.query}" (${k.impressions} impressions, pos ${k.position})</li>`; });
       html += '</ul>';
     }
     if (d.lostKeywords.length > 0) {
       html += '<h2>Lost Keywords</h2><ul>';
       d.lostKeywords.forEach(k => { html += `<li>"${k.query}" (was ${k.lastImpressions} impressions, pos ${k.lastPosition})</li>`; });
       html += '</ul>';
     }

     // Content gaps callout
     if (d.contentGapCount > 0) {
       html += `<h2>Content Gaps</h2><p>${d.contentGapCount} content gaps detected (see TG-98 report for details).</p>`;
     }

     html += '<hr><p>Generated by TotalGuard SEO Monitor. Data from Google Search Console (2-day lag).</p>';

     return [{ json: {
       to_email: 'totalguardllc@gmail.com',
       subject: `TG SEO Weekly: ${o.totalClicks} clicks (${o.clickChange >= 0 ? '+' : ''}${o.clickChange}), ${d.topGainers.length} gainers, ${d.topLosers.length} losers`,
       html_body: html,
       workflow_name: 'TG-102',
       email_type: 'seo_weekly_summary'
     }}];

  -> Execute Sub-workflow "Send via TG-95"
     workflowId: IUDLrQrAkcLFLsIC (TG-95)
     Input: { to_email, subject, html_body, workflow_name, email_type }
```
