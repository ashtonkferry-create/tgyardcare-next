   - Code:
     ```javascript
     const weather = $input.first().json;
     const temp = weather.main?.temp || 70;
     const conditions = (weather.weather || []).map(w => w.main.toLowerCase());
     const windSpeed = weather.wind?.speed || 0;
     const snow = weather.snow?.['1h'] || weather.snow?.['3h'] || 0;
     const rain = weather.rain?.['1h'] || weather.rain?.['3h'] || 0;
     const description = (weather.weather || []).map(w => w.description).join(', ');

     let triggers = [];

     // Heavy snow (>2 inches expected or active snow with low temp)
     if (snow > 0 || (conditions.includes('snow') && temp < 35)) {
       triggers.push({ type: 'snow', subject: 'Snow alert! TotalGuard has you covered', service: 'Snow Removal', urgency: 'high' });
     }

     // Heatwave (>90F)
     if (temp > 90) {
       triggers.push({ type: 'heatwave', subject: 'Beat the heat — protect your lawn', service: 'Lawn Care', urgency: 'medium' });
     }

     // Severe storm (high wind + rain)
     if (windSpeed > 30 || (rain > 1 && windSpeed > 20)) {
       triggers.push({ type: 'storm', subject: 'Storm cleanup? We\'re ready to help', service: 'Storm Cleanup', urgency: 'high' });
     }

     // Freeze warning (temp < 28F in fall/spring)
     const month = new Date().getMonth() + 1;
     if (temp < 28 && (month >= 3 && month <= 5 || month >= 9 && month <= 11)) {
       triggers.push({ type: 'freeze', subject: 'Freeze warning — protect your landscaping', service: 'Winterization', urgency: 'medium' });
     }

     if (triggers.length === 0) return []; // No weather event — workflow ends

     return triggers.map(t => ({
       json: {
         ...t,
         current_temp: temp,
         conditions: description,
         wind_speed: windSpeed
       }
     }));
     ```

3. **Identify Dormant** (code node, typeVersion 2)
   - ID: identify-dormant
   - Code:
     ```javascript
     const customers = $input.all().map(i => i.json);
     const now = Date.now();
     const NINETY_DAYS = 90 * 24 * 60 * 60 * 1000;
     const THIRTY_DAYS = 30 * 24 * 60 * 60 * 1000;

     const dormant = [];
     for (const c of customers) {
       const lastActivity = new Date(c.updated_at || c.created_at || '2020-01-01').getTime();
       const daysSince = Math.floor((now - lastActivity) / (24*60*60*1000));

       // Skip if not dormant (< 90 days)
       if (now - lastActivity < NINETY_DAYS) continue;

       // Skip if win-back already sent in last 30 days
       if (c.winback_sent_at) {
         const lastWinback = new Date(c.winback_sent_at).getTime();
         if (now - lastWinback < THIRTY_DAYS) continue;
       }

       // Must have email or phone
       if (!c.email && !c.phone) continue;

       dormant.push({ json: { ...c, days_dormant: daysSince } });
     }

     return dormant;
     ```
