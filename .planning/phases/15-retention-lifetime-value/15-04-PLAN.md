3. **Analyze Cross-Sell Opportunities** (code node, typeVersion 2)
   - ID: analyze-crosssell
   - Code:
     ```javascript
     const customers = $input.all().map(i => i.json);

     // Service bundles — if customer uses X, recommend Y
     const crossSellMap = {
       'mowing': ['fertilization', 'aeration', 'spring cleanup', 'fall cleanup'],
       'lawn mowing': ['fertilization', 'aeration', 'spring cleanup', 'fall cleanup'],
       'lawn care': ['mulching', 'garden bed care', 'bush trimming', 'gutter cleaning'],
       'fertilization': ['aeration', 'herbicide', 'overseeding'],
       'aeration': ['fertilization', 'overseeding'],
       'spring cleanup': ['mulching', 'garden bed care', 'fertilization'],
       'fall cleanup': ['leaf removal', 'gutter cleaning', 'snow removal'],
       'mulching': ['garden bed care', 'bush trimming', 'weeding'],
       'snow removal': ['gutter guard installation', 'gutter cleaning'],
       'gutter cleaning': ['gutter guard installation'],
       'hardscaping': ['mulching', 'garden bed care', 'lawn care']
     };

     const results = [];
     for (const c of customers) {
       if (!c.email) continue;
       const currentService = (c.service || '').toLowerCase();
       const recommendations = crossSellMap[currentService] || ['fertilization', 'aeration'];

       // Pick top 2 recommendations
       const topRecs = recommendations.slice(0, 2);

       results.push({
         json: {
           lead_id: c.id,
           first_name: c.first_name,
           email: c.email,
           current_service: c.service,
           recommended_services: topRecs,
           rec1: topRecs[0] || 'fertilization',
           rec2: topRecs[1] || 'aeration'
         }
       });
     }

     return results;
     ```
