4. **Process Stale Leads** (code node, typeVersion 2)
   - ID: process-stale-leads
   - Code logic:
     ```javascript
     const leads = $('Fetch Uncontacted Leads').all().map(i => i.json);
     const alertHistory = $('Fetch Alert History').all().map(i => i.json);

     // Build set of already-alerted: "lead_id|alert_type"
     const alerted = new Set();
     for (const a of alertHistory) {
       if (a.lead_id && a.alert_type) alerted.add(`${a.lead_id}|${a.alert_type}`);
     }

     // Check business hours (CT = UTC-5/6)
     const now = new Date();
     const ctHour = new Date(now.toLocaleString('en-US', { timeZone: 'America/Chicago' })).getHours();
     if (ctHour < 8 || ctHour >= 21) return []; // Outside business hours

     const results = [];
     for (const lead of leads) {
       const ageMs = Date.now() - new Date(lead.created_at).getTime();
       const ageHours = ageMs / (1000 * 60 * 60);

       let alertLevel = null;
       let severity = '';
       let emoji = '';

       if (ageHours >= 8) {
         alertLevel = 'speed_to_lead_8h';
         severity = 'CRITICAL';
         emoji = '🚨🚨🚨';
       } else if (ageHours >= 4) {
         alertLevel = 'speed_to_lead_4h';
         severity = 'WARNING';
         emoji = '⚠️⚠️';
       } else if (ageHours >= 2) {
         alertLevel = 'speed_to_lead_2h';
         severity = 'ALERT';
         emoji = '⏰';
       }

       if (!alertLevel) continue;
       if (alerted.has(`${lead.id}|${alertLevel}`)) continue;

       const name = `${lead.first_name || ''} ${lead.last_name || ''}`.trim() || 'Unknown';
       const phone = lead.phone || 'No phone';
       const service = lead.service || 'Not specified';
       const hoursAgo = Math.round(ageHours * 10) / 10;

       const msg = `${emoji} SPEED TO LEAD ${severity}\n\n` +
         `Lead waiting ${hoursAgo} hours!\n\n` +
         `👤 ${name}\n` +
         `📱 ${phone}\n` +
         `🔧 ${service}\n` +
         `🔥 Score: ${lead.lead_score || 0} (${lead.lead_tier || 'unknown'})\n\n` +
         `📞 Call NOW: tel:${(phone || '').replace(/[^+\d]/g, '')}`;

       results.push({
         json: {
           lead_id: lead.id,
           alert_type: alertLevel,
           message: msg,
           severity,
           hours_waiting: hoursAgo
         }
       });
     }
     return results;
     ```
